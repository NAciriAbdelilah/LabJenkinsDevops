pipeline{
    agent any
    tools {
      maven 'MVN_HOME'
    }
    environment {
      DOCKER_TAG = getVersion()
    }
    stages{
        stage('mvn Build'){
            steps{
                sh "mvn -v"
            }
        }

        stage('SCM'){
            steps{
                git credentialsId: 'github', 
                    url: 'https://github.com/NAciriAbdelilah/LabJenkinsDevops.git'
            }
        }
        
        stage('Maven Build'){
            steps{
                sh "mvn clean package"
            }
        }
        
        stage('Docker Build'){
            steps{
                sh "docker build . -t naciriTestingLab:${DOCKER_TAG}"
            }
        }
        
    }
}

def getVersion(){
    def commitHash = sh label: '', returnStdout: true, script: 'git rev-parse --short HEAD'
    return commitHash
}
